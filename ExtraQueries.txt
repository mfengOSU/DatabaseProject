SELECT ITEM.Title
FROM ITEM, AUDIOBOOK, NARRATOR
WHERE 
    ITEM.ItemNo = AUDIOBOOK.BookNo AND
    AUDIOBOOK.AuthorId = NARRATOR.NarratorId AND
    NARRATOR.Name = 'Verla Lannen'
;

SELECT AVG(LIBRARIAN.Salary)
FROM LIBRARY, LIBRARIAN
WHERE (LIBRARY.City, LIBRARY.State) = (LIBRARIAN.City, LIBRARIAN.State) 
GROUP BY LIBRARY.City
;

SELECT ITEM.Title
FROM ITEM, MOVIE, ACTOR, MOVIE_ACTORS
WHERE ITEM.ItemNo = MOVIE.MovieNo AND
    MOVIE.MovieNo = MOVIE_ACTORS.MovieNo AND
    ACTOR.ActorId = MOVIE_ACTORS.ActorId AND
    ACTOR.Name = 'Alberto Livoir'
;
