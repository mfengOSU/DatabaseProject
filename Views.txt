CREATE VIEW PATRON_CHECKOUTS_VIEW AS
SELECT PATRON.FName, PATRON.LName, COUNT(*) as NumberCheckedOut
FROM ITEM, PATRON, PATRON_CHECKOUTS
WHERE 
PATRON_CHECKOUTS.ItemNo = ITEM.ItemNo AND
PATRON.PatronId = PATRON_CHECKOUTS.PatronId 
GROUP BY PATRON.PatronId;


CREATE VIEW LIBRARY_LIBRARIANS_VIEW AS
SELECT LIBRARY.City, LIBRARY.State, Count(*) as NumberLibrarians, AVG(LIBRARIAN.Salary) as AvgSalary
FROM LIBRARY, LIBRARIAN
WHERE (LIBRARY.City, LIBRARY.State) = (LIBRARIAN.City, LIBRARIAN.State) 
GROUP BY LIBRARY.City;
